7   /**
8    * Abstract representation of a minegrid.
9    *
10   * @author Saskeli
11   */
12  public class GameGrid implements Serializable {
13      /**
14       * The actual game grid. A square array of tiles
15       */
16      private final Tile[][] tiles;
17      /**
18       * Indicates if the game has been started.
19       * I.e. if the tile array has been populated.
20       */
21      private boolean started;
22      /**
23       * Number of mines specified by the constructor.
24       * Note: May be higher than the actual number of mines
25       * in the array.
26       */
27      private final int mines;
28      /**
29       * Coordinate of the bottom right tile in the array.
30       * Usefull for tile validation.
31       */ 
32      private final Coordinate max;
33      /**
34       * The type of game used for creating this grid.
35       */
36      private final GameType gameType;
37  
38      /**
39       * Abstract representation of a minegrid.
40       * 
41       * @param width     the width of the grid to be created
42       * @param height    the hight of the grid to be created
43       * @param mines     the desired mine count
44       */
45      public GameGrid(int width, int height, int mines) {
46          gameType = GameType.CUSTOM;
47          this.mines = mines;
48          tiles = new Tile[Math.max(1, height)][Math.max(1, width)];
49          this.started = false;
50          this.max = new Coordinate(height - 1, width - 1);
51      }
52      
53      /**
54       * Abstract representation of a minegrid.
55       * 
56       * @param tiles  a square array of tiles.
57       */
58      public GameGrid(Tile[][] tiles) {
59          gameType = GameType.CUSTOM;
60          this.tiles = tiles;
61          this.started = true;
62          this.mines = -1;
63          this.max = new Coordinate(this.tiles.length - 1, this.tiles[0].length - 1);
64      }
65      
66      /**
67       * Abstract representation of a minegrid.
68       * Note: With the CUSTOM game type. A HARD game will be started instead.
69       * 
70       * @param gameType the type of game to create
71       */
72      public GameGrid(GameType gameType) {
73          if (gameType == GameType.EASY) {
74              this.gameType = gameType;
75              this.mines = 10;
76              this.tiles = new Tile[9][9];
77              this.started = false;
78              this.max = new Coordinate(8, 8);
79          } else if (gameType == GameType.NORMAL) {
80              this.gameType = gameType;
81              this.mines = 40;
82              this.tiles = new Tile[16][16];
83              this.started = false;
84              this.max = new Coordinate(15, 15);
85          } else {
86              this.gameType = GameType.HARD;
87              this.mines = 99;
88              this.tiles = new Tile[16][30];
89              this.started = false;
90              this.max = new Coordinate(15, 29);
91          }
92      }
93  
94      public GameType getGameType() {
95          return gameType;
96      }
97  
98      /**
99       * get the number of mines that are known to be adjacent to 
100      * the given coordinate
101      * 
102      * @param coord the coodinate to check around
103      * @return      0 if the game has not started, 9 if the tile is a mine, and 0-8 otherwise.
104      */
105     public int getValue(Coordinate coord) {
106         if (!started) {
107             return 0;
108         }
109         if (hasMine(coord)) {
110             return 9;
111         }
112         return surroundingMines(coord);
113     }
114     
115     /**
116      * Check if a given coordinate is valid.
117      * I.e. that the coordinate is withing the curren game.
118      * 
119      * @param coord the coordinate to check
120      * @return      true if the coordinate is on the grid.
121      */
122     public boolean isValid(Coordinate coord) {
123         return coord.isValid(max);
124     }
125 
126     /**
127      * Get the number of mines known to be adjacent to a given coordinate.
128      * 
129      * @param coord Coordinate to check around
130      * @return      Number of mines 0-8.
131      */
132     private int surroundingMines(Coordinate coord) {
133         int adjacentMines = 0;
134         for (Coordinate coordinate : coord.getAdjacentCoordinates(max)) {
135             if (tiles[coordinate.getRow()][coordinate.getColumn()].isMine()) {
136                 adjacentMines += 1;
137             }
138         }
139         return adjacentMines;
140     }
141     
142     /**
143      * Check if there is a mine at a given coordiante.
144      * 
145      * @param coord  the coordinate to check
146      * @return       true if there is a mine at the given coordinate
147      */
148     public boolean hasMine(Coordinate coord) {
149         try {
150             return tiles[coord.getRow()][coord.getColumn()].isMine();
151         } catch (ArrayIndexOutOfBoundsException e) {
152             return false;
153         }
154     }
155 
156     /**
157      * Clears the tile at the given coordinate.
158      * Esentially clicks the tile.
159      * 
160      * @param coord  the oordinate to clear
161      * @return       false if the cleared tile was a mine
162      */
163     public boolean clear(Coordinate coord) {
164         if (!coord.isValid(max) || isCleared(coord) || isFlagged(coord)) {
165             return true;
166         }
167         if (!started) {
168             generateTiles(coord);
169             started = true;
170         }
171         Tile tile = tiles[coord.getRow()][coord.getColumn()];
172         tile.setChecked(true);
173         if (tile.isMine()) {
174             clearMines();
175             return false;
176         }
177         if (surroundingMines(coord) == 0) {
178             clearSurroundingTiles(coord);
179         }
180         return true;
181     }
182 
183     /**
184      * Populates the tile array.
185      * 
186      * @param coord  the coordinate that can't be a mine
187      */
188     private void generateTiles(Coordinate coord) {
189         List<Boolean> mineList = new ArrayList<>();
190         List<Coordinate> specialSquares = coord.getAdjacentCoordinates(max);
191         specialSquares.add(coord);
192         populateList(mineList, specialSquares.size());
193         populateTileGrid(mineList, specialSquares);
194         int extraMines = mines - (tiles.length * tiles[0].length - specialSquares.size());
195         if (extraMines > 0) {
196             addMinesToSpecials(extraMines, coord);
197         }
198     }
199 
200     /**
201      * Generates a randomized list of mines
202      * 
203      * @param mineList        the list of to populate
204      * @param specialSquares  the minimum number of squares to leave clear
205      */
206     private void populateList(List<Boolean> mineList, int specialSquares) {
207         for (int i = 0; i < (tiles.length * tiles[0].length) - specialSquares; i++) {
208             if (i < this.mines) {
209                 mineList.add(true);
210             } else {
211                 mineList.add(false);
212             }
213         }
214         Collections.shuffle(mineList);
215     }
216 
217     /**
218      * Maps the list of booleans to the tile array.
219      * 
220      * @param mineList        the list of booleans signifying mines
221      * @param specialSquares  the squares to leave clear
222      */
223     private void populateTileGrid(List<Boolean> mineList, List<Coordinate> specialSquares) {
224         for (int i = 0; i < tiles.length; i++) {
225             for (int j = 0; j < tiles[0].length; j++) {
226                 Coordinate coord = new Coordinate(i, j);
227                 if (!specialSquares.contains(coord)) {
228                     tiles[i][j] = new Tile(mineList.get(0), false);
229                     mineList.remove(0);
230                 } else {
231                     tiles[i][j] = new Tile(false, false);
232                 }
233             }
234         }
235     }
236 
237     /**
238      * Used to populate squares around the initially cleared square on 
239      * game start if there are more mines than initially populated.
240      * 
241      * @param minesToAdd   the number of mines to add
242      * @param aroundCoord  the coordinate to leave clear
243      */
244     public void addMinesToSpecials(int minesToAdd, Coordinate aroundCoord) {
245         List<Boolean> mineList = new ArrayList<>();
246         List<Coordinate> specialSquares = aroundCoord.getAdjacentCoordinates(max);
247         for (int j = 0; j < specialSquares.size(); j++) {
248             if (j < minesToAdd) {
249                 mineList.add(true);
250             } else {
251                 mineList.add(false);
252             }
253         }
254         Collections.shuffle(mineList);
255         for (int j = 0; j < specialSquares.size(); j++) {
256             tiles[specialSquares.get(j).getRow()][specialSquares.get(j).getColumn()].setMine(mineList.get(j));
257         }
258     }
259 
260     /**
261      * Sets the state of all tiles with mines to cleared.
262      * Usefull for clearing all mines on game over.
263      */
264     public void clearMines() {
265         for (Tile[] tileRow : tiles) {
266             for (Tile tile : tileRow) {
267                 if (tile.isMine()) {
268                     tile.setChecked(true);
269                 }
270             }
271         }
272     }

274     @Override
275     public String toString() {
276         StringBuilder stringBuilder = new StringBuilder();
277         for (Tile[] tileRow : tiles) {
278             for (Tile tile : tileRow) {
279                 stringBuilder.append(tile.toString());
280             }
281             stringBuilder.append("\n");
282         }
283         return stringBuilder.toString();
284     }

286     /**
287      * Get the number of tiles left to clear. 
288      * I.e. tiles that are not cleared and are not mines.
289      * 
290      * @return  the number of tiles left to clear
291      */
292     public int leftToClear() {
293         if (!started) {
294             return -1;
295         }
296         int count = 0;
297         for (Tile[] tileRow : tiles) {
298             for (Tile tile : tileRow) {
299                 if (tile.isMine() | tile.isChecked()) {
300                     continue;
301                 }
302                 count++;
303             }
304         }
305         return count;
306     }

344      * 
345      * @param coordinate  the coordinate to check

348     public boolean isFlagged(Coordinate coordinate) {
349         if (!started) {
350             return false;

352         return tiles[coordinate.getRow()][coordinate.getColumn()].isFlagged();
353     }

355     /**
356      * Toggles the flag of the tile at the specified coordinate
357      * 
358      * @param coordinate  the coordinate for the tile to toggle
359      */
360     public void toggleFlag(Coordinate coordinate) {
361         if (!started || isCleared(coordinate)) {
362             return;
363         }
364         tiles[coordinate.getRow()][coordinate.getColumn()].toggleFlag();
365     }
366 
367     /**
368      * Clears the non-flagged tiles around the specified coordinate
369      * if the value of the tile specified by the coordinate equals 
370      * the number of flags adjacent to the tile.
371      * 
372      * @param coordinate  the coordinate to clear around
373      */
374     public void clearSurrounding(Coordinate coordinate) {
375         if (surroundingMines(coordinate) == surroundingFlags(coordinate)) {
376             clearSurroundingTiles(coordinate);
377         }
378     }
379 
380     /**
381      * Get the number of flagged tiles surrounding the tile specified
382      * by the given coordinate
383      * 
384      * @param coordinate  the coordinate to check around
385      * @return            the number of flags surrounding the coordinate
386      */
387     private int surroundingFlags(Coordinate coordinate) {
388         int adjacentFlags = 0;
389         for (Coordinate coord : coordinate.getAdjacentCoordinates(max)) {
390             if (tiles[coord.getRow()][coord.getColumn()].isFlagged()) {
391                 adjacentFlags += 1;
392             }
393         }
394         return adjacentFlags;
395     }
396 
397     /**
398      * Calls clear for all the coordinates surrounding the specified coordiante
399      * 
400      * @param coord  the coordinate to clear around
401      */
402     private void clearSurroundingTiles(Coordinate coord) {
403         for (Coordinate coordinate : coord.getAdjacentCoordinates(max)) {
404             clear(coordinate);
405         }
406     }
