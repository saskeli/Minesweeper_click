1   package minesweeper.logic;
2   
3   import minesweeper.util.Coordinate;
4   import java.util.*;
5   
6   /**
7    * Abstract representation of a minegrid.
8    * 
9    * @author Saskeli
10   */
11  public class GameGrid {
12  
13      private final Tile[][] tiles;
14      private boolean started;
15      private final int mines;
16      private final Coordinate max;
17  
18      public GameGrid(int width, int height, int mines) {
19          this.mines = mines;
20          tiles = new Tile[Math.max(1, height)][Math.max(1, width)];
21          this.started = false;
22          this.max = new Coordinate(height - 1, width - 1);
23      }
24  
25      public GameGrid(Tile[][] tiles) {
26          this.tiles = tiles;
27          this.started = true;
28          this.mines = -1;
29          this.max = new Coordinate(this.tiles.length - 1, this.tiles[0].length - 1);
30      }
31  
32      public int getValue(Coordinate coord) {
33          if (!started) {
34              return -1;
35          }
36          if (hasMine(coord)) {
37              return 9;
38          }
39          return surroundingMines(coord);
40      }
41  
42      public boolean isValid(Coordinate coord) {
43          return coord.isValid(max);
44      }
45  
46      public int surroundingMines(Coordinate coord) {
47          int adjacentMines = 0;
48          for (Coordinate coordinate : coord.getAdjacentCoordinates(max)) {
49              if (tiles[coordinate.getRow()][coordinate.getColumn()].isMine()) {
50                  adjacentMines += 1;
51              }
52          }
53          return adjacentMines;
54      }
55  
56      public boolean hasMine(Coordinate coord) {
57          try {
58              return tiles[coord.getRow()][coord.getColumn()].isMine();
59          } catch (ArrayIndexOutOfBoundsException e) {
60              return false;
61          }
62      }
63  
64      public boolean clear(Coordinate coord) {
65          if (!coord.isValid(max)) {
66              return true;
67          }
68          if (!started) {
69              generateTiles(coord);
70              started = true;
71          }
72          Tile tile = tiles[coord.getRow()][coord.getColumn()];
73          tile.setChecked(true);
74          if (tile.isMine()) {
75              clearMines();
76              return false;
77          }
78          if (surroundingMines(coord) == 0) {
79              for (Coordinate coordinate : coord.getAdjacentCoordinates(max)) {
80                  if (!tiles[coordinate.getRow()][coordinate.getColumn()].isChecked()) {
81                      clear(coordinate);
82                  }
83              }
84          }
85          return true;
86      }
87  
88      public void generateTiles(Coordinate coord) {
89          List<Boolean> mineList = new ArrayList<>();
90          List<Coordinate> specialSquares = coord.getAdjacentCoordinates(max);
91          specialSquares.add(coord);
92          populateList(mineList, specialSquares.size());
93          populateTileGrid(mineList, specialSquares);
94          int extraMines = mines - (tiles.length * tiles[0].length - specialSquares.size());
95          if (extraMines > 0) {
96              addMinesToSpecials(extraMines, coord);
97          }
98      }
99  
100     public void populateList(List<Boolean> mineList, int specialSquares) {
101         for (int i = 0; i < (tiles.length * tiles[0].length) - specialSquares; i++) {
102             if (i < this.mines) {
103                 mineList.add(true);
104             } else {
105                 mineList.add(false);
106             }
107         }
108         Collections.shuffle(mineList);
109     }
110 
111     public void populateTileGrid(List<Boolean> mineList, List<Coordinate> specialSquares) {
112         for (int i = 0; i < tiles.length; i++) {
113             for (int j = 0; j < tiles[0].length; j++) {
114                 Coordinate coord = new Coordinate(i, j);
115                 if (!isOnList(coord, specialSquares)) {
116                     tiles[i][j] = new Tile(mineList.get(0), false);
117                     mineList.remove(0);
118                 } else {
119                     tiles[i][j] = new Tile(false, false);
120                 }
121             }
122         }
123     }
124 
125     public void addMinesToSpecials(int minesToAdd, Coordinate aroundCoord) {
126         List<Boolean> mineList = new ArrayList<>();
127         List<Coordinate> specialSquares = aroundCoord.getAdjacentCoordinates(max);
128         for (int j = 0; j < specialSquares.size(); j++) {
129             if (j < minesToAdd) {
130                 mineList.add(true);
131             } else {
132                 mineList.add(false);
133             }
134         }
135         Collections.shuffle(mineList);
136         for (int j = 0; j < specialSquares.size(); j++) {
137             tiles[specialSquares.get(j).getRow()][specialSquares.get(j).getColumn()].setMine(mineList.get(j));
138         }
139     }
140 
141     public void clearMines() {
142         for (Tile[] tileRow : tiles) {
143             for (Tile tile : tileRow) {
144                 if (tile.isMine()) {
145                     tile.setChecked(true);
146                 }
147             }
148         }
149     }
150 
151     @Override
152     public String toString() {
153         StringBuilder stringBuilder = new StringBuilder();
154         for (Tile[] tileRow : tiles) {
155             for (Tile tile : tileRow) {
156                 stringBuilder.append(tile.toString());
157             }
158             stringBuilder.append("\n");
159         }
160         return stringBuilder.toString();
161     }
162 
163     public boolean isOnList(Coordinate coordinate, List<Coordinate> specialSquares) {
164         return specialSquares.contains(coordinate);
165     }
166 
167     public int leftToClear() {
168         if (!started) {
169             return -1;
170         }
171         int count = 0;
172         for (Tile[] tileRow : tiles) {
173             for (Tile tile : tileRow) {
174                 if (tile.isMine() | tile.isChecked()) {
175                     continue;
176                 }
177                 count++;
178             }
179         }
180         return count;
181     }
182 
183     public int getWidth() {
184         return tiles[0].length;
185     }
186 
187     public int getHeight() {
188         return tiles.length;
189     }
190 
191     public boolean isCleared(Coordinate coord) {
192         if (!started) {
193             return false;
194         }
195         return tiles[coord.getRow()][coord.getColumn()].isChecked();
196     }
197 
198     public int getMines() {
199         return mines;
200     }
201 
202     public boolean isFlagged(Coordinate coordinate) {
203         if (!started) {
204             return false;
205         }
206         return tiles[coordinate.getRow()][coordinate.getColumn()].isFlagged();
207     }
208 
209     public void toggleFlag(Coordinate coordinate) {
210         if (!started) {
211             return;
212         }
213         tiles[coordinate.getRow()][coordinate.getColumn()].toggleFlag();
214     }
215 }
