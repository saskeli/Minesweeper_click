1   package minesweeper.gui;
2   
3   import java.awt.Component;
4   import java.awt.GridLayout;
5   import javax.swing.JPanel;
6   import minesweeper.eventhandlers.SquareClickListener;
7   import minesweeper.main.Game;
8   import minesweeper.util.Coordinate;
9   import minesweeper.util.GameType;
10  
11  /**
12   * Class describing minefield in JPanel form
13   *
14   * @author Saskeli
15   */
16  public class GamePanel extends JPanel {
17      /**
18       * Game logic interface
19       */
20      private final Game game;
21      /**
22       * Gui parent
23       */
24      private final Gui gui;
25      /**
26       * Panel to be updated with game stats
27       */
28      private StatPanel statPanel = null;
29      /**
30       * Action listener to be used with all Squares
31       */
32      private final SquareClickListener squareClickListener;
33  
34      /**
35       * Class describing minefield in JPanel form
36       * 
37       * @param game  Game logic interface
38       * @param gui   The parent GUI.
39       */
40      public GamePanel(Game game, Gui gui) {
41          this.game = game;
42          this.gui = gui;
43          squareClickListener = new SquareClickListener(game, this);
44          populateGrid();
45          updateGrid();
46      }
47      
48      /**
49       * Refreshes the visual representation of the game
50       */
51      public final void updateGrid() {
52          for (Component component : this.getComponents()) {
53              Square s = (Square) component;
54              s.update();
55          }
56          if (statPanel != null) {
57              statPanel.setActions(game.getActionCount());
58              statPanel.setRemainingCount(game.getRemainingTiles());
59          }
60          if (game.getRemainingTiles() == 0) {
61              gui.checkScore(game.getGameType(), game.getActionCount());
62          }
63      }
64      
65      public void setStatPanel(StatPanel statPanel) {
66          this.statPanel = statPanel;
67      }
68  
69      /**
70       * Starts a new game with the same parameters as the original
71       * and updates the visual representation
72       */
73      public void startNewGame() {
74          game.newGame();
75          updateGrid();
76      }
77  
78      /**
79       * Starts an new custom game and updates the visual representation
80       * 
81       * @param width     Width of the game to be created
82       * @param height    height of the game to be created
83       * @param mines     Number of mines for the new game
84       */
85      public void startNewGame(int width, int height, int mines) {
86          game.newGame(width, height, mines);
87          removeAll();
88          populateGrid();
89          gui.rePack();
90      }
91      
92      /**
93       * Starts a new game with one of the standart settings.
94       * Note: Custom will start a hard game
95       * 
96       * @param gameType  Type of game to be started
97       */
98      public void startNewGame(GameType gameType) {
99          game.newGame(gameType);
100         removeAll();
101         populateGrid();
102         gui.rePack();
103     }
104     
105     /**
106      * Add the sqares to the gird and the action listener 
107      * to the sqares
108      */
109     private void populateGrid() {
110         setLayout(new GridLayout(game.gameHeight(), game.gameWidth()));
111         for (int i = 0; i < game.gameHeight(); i++) {
112             for (int j = 0; j < game.gameWidth(); j++) {
113                 add(new Square(game, new Coordinate(i, j), squareClickListener));
114             }
115         }
116         if (statPanel != null) {
117             statPanel.setActions(game.getActionCount());
118             statPanel.setRemainingCount(game.getRemainingTiles());
119         }
120     }
121 }
