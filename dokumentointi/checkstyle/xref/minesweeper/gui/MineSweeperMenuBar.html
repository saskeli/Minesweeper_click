1   package minesweeper.gui;
2   
3   import java.awt.PopupMenu;
4   import java.awt.event.KeyEvent;
5   import javax.swing.*;
6   import minesweeper.eventhandlers.ButtonActionListener;
7   
8   /**
9    * Menu bar for minesweeper_click
10   * 
11   * @author dongesa
12   */
13  class MineSweeperMenuBar extends JMenuBar {
14      
15      /**
16       * Menu bar for minesweeper_click
17       * 
18       * @param buttonActionListener  the action listener to use for menu buttons
19       */
20      public MineSweeperMenuBar(ButtonActionListener buttonActionListener) {
21          add(newMenu(buttonActionListener));
22          add(scoreMenu(buttonActionListener));
23      }
24  
25      /**
26       * Create the new game menu
27       * 
28       * @param buttonActionListener  the action listener to use for menu buttons
29       * @return                      the new game menu
30       */
31      private JMenu newMenu(ButtonActionListener buttonActionListener) {
32          JMenu newMenu = new JMenu("New");
33          newMenu.setMnemonic(KeyEvent.VK_N);
34          newMenu.getAccessibleContext().setAccessibleDescription("Create new game");
35          newMenu.add(menuItem("Custom", KeyEvent.VK_C, "CustomButton", "Create custom game", buttonActionListener));
36          newMenu.add(menuItem("Easy", KeyEvent.VK_E, "EasyButton", "Create new easy game", buttonActionListener));
37          newMenu.add(menuItem("Normal", KeyEvent.VK_N, "NormalButton", "Create new normal game", buttonActionListener));
38          newMenu.add(menuItem("Hard", KeyEvent.VK_H, "HardButton", "Create new hard game", buttonActionListener));
39          return newMenu;
40      }
41  
42      /**
43       * Create the high score menu
44       * 
45       * @param buttonActionListener  the action listener to use for menu buttons
46       * @return                      the high score menu
47       */
48      private JMenu scoreMenu(ButtonActionListener buttonActionListener) {
49          JMenu scoreMenu = new JMenu("Scores");
50          scoreMenu.setMnemonic(KeyEvent.VK_S);
51          scoreMenu.getAccessibleContext().setAccessibleDescription("High scores and reset");
52          scoreMenu.add(menuItem("High Scores", KeyEvent.VK_H, "ScoreButton", "Show high scores", buttonActionListener));
53          scoreMenu.add(menuItem("Clear scores", KeyEvent.VK_C, "ResetButton", "Clear high scores", buttonActionListener));
54          return scoreMenu;
55      }
56      
57      /**
58       * Create a JMenuItem
59       * 
60       * @param text                  the button text
61       * @param mnemonic              the hotkey for the button
62       * @param name                  the name of the button
63       * @param tooltip               the tooltip for the button
64       * @param buttonActionListener  the action listener to use for menu buttons
65       * @return                      the new JMenuItem
66       */
67      private JMenuItem menuItem(String text, int mnemonic, String name, String tooltip, ButtonActionListener buttonActionListener) {
68          JMenuItem item = new JMenuItem(text, mnemonic);
69          item.setName(name);
70          item.getAccessibleContext().setAccessibleDescription(tooltip);
71          item.addActionListener(buttonActionListener);
72          return item;
73      }
74  }
