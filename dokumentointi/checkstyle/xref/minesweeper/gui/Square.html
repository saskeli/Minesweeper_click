1   package minesweeper.gui;
2   
3   import java.awt.Color;
4   import java.awt.Dimension;
5   import javax.swing.*;
6   import minesweeper.eventhandlers.SquareClickListener;
7   import minesweeper.main.Game;
8   import minesweeper.util.Coordinate;
9   
10  /**
11   * Representation of a minesweeper square for GUI
12   * 
13   * @author Saskeli
14   */
15  public class Square extends JLabel {
16      private Game game;
17      private Coordinate coordinate;
18      private final Color defaultColor;
19  
20      public Square(Game game, Coordinate coordinate, SquareClickListener listener) {
21          super("");
22          this.game = game;
23          this.coordinate = coordinate;
24          addMouseListener(listener);
25          setOpaque(true);
26          setHorizontalAlignment(JLabel.CENTER);
27          setVerticalAlignment(JLabel.CENTER);
28          setPreferredSize(new Dimension(25, 25));
29          setBorder(BorderFactory.createRaisedBevelBorder());
30          defaultColor = this.getBackground();
31      }
32  
33      public Coordinate getCoordinate() {
34          return coordinate;
35      }
36  
37      public void update() {
38          int tileState = game.getTileState(coordinate);
39          if (game.isFlagged(coordinate) && tileState == -1) {
40              return;
41          }
42          if (tileState == -1) {
43              setBorder(BorderFactory.createRaisedBevelBorder());
44              setBackground(defaultColor);
45              setText("");
46          } else if (tileState == 0) {
47              setBackground(Color.WHITE);
48              setText("");
49              setBorder(BorderFactory.createLineBorder(Color.GRAY));
50          } else if (tileState == 9) {
51              setBackground(Color.red);
52              setText("*");
53              setBorder(BorderFactory.createLineBorder(Color.GRAY));
54          } else {
55              setBackground(Color.WHITE);
56              setText("" + tileState);
57              setBorder(BorderFactory.createLineBorder(Color.GRAY));
58          }
59      }
60  
61      public void toggleFlag() {
62          game.toggleFlag(coordinate);
63          if (isFlagged()) {
64              setBackground(defaultColor);
65          } else {
66              setBackground(Color.PINK);
67          }
68      }
69  
70      public boolean isFlagged() {
71          return game.isFlagged(coordinate);
72      }
73  
74      public boolean isCleared() {
75          return (game.getTileState(coordinate) != -1);
76      }
77  }
