1   package minesweeper.gui;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Dimension;
5   import javax.swing.*;
6   import minesweeper.eventhandlers.ButtonActionListener;
7   
8   /**
9    * Panel for the minwsweeper GUI that shows stats and provides the 
10   * new game button.
11   * 
12   * @author dongesa
13   */
14  public class StatPanel extends JPanel {
15      private final JLabel clickCountLabel;
16      private final JLabel toClearLabel;
17      
18      public StatPanel(ButtonActionListener buttonActionListener) {
19          Dimension d = new Dimension(40, 10);
20          
21          toClearLabel = new JLabel("-1", JLabel.CENTER);
22          toClearLabel.setPreferredSize(d);
23          add(toClearLabel, BorderLayout.WEST);
24          
25          JButton reStartButton = new JButton("New game");
26          reStartButton.setName("NewButton");
27          reStartButton.addActionListener(buttonActionListener);
28          add(reStartButton, BorderLayout.CENTER);
29          
30          clickCountLabel = new JLabel("0", JLabel.CENTER);
31          clickCountLabel.setPreferredSize(d);
32          add(clickCountLabel, BorderLayout.EAST);
33      }
34  
35      public void setActions(int actionCount) {
36          clickCountLabel.setText(actionCount + "");
37      }
38  
39      public void setRemainingCount(int remainingTiles) {
40          toClearLabel.setText(remainingTiles + "");
41      }
42  }
