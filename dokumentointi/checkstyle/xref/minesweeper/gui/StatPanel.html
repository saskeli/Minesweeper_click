1   package minesweeper.gui;
2   
3   import java.awt.BorderLayout;
4   import java.awt.Dimension;
5   import javax.swing.*;
6   import minesweeper.eventhandlers.ButtonActionListener;
7   
8   /**
9    * Panel for the minwsweeper GUI that shows stats and provides the 
10   * new game button.
11   * 
12   * @author dongesa
13   */
14  public class StatPanel extends JPanel {
15      /**
16       * Label that tracks number of actions.
17       */
18      private final JLabel clickCountLabel;
19      /**
20       * Lable that tracks number of tiles to clear.
21       */
22      private final JLabel toClearLabel;
23      
24      /**
25       * Panel for the minwsweeper GUI that shows stats and provides the 
26       * new game button.
27       * 
28       * @param buttonActionListener  the action listener for the new game button
29       */
30      public StatPanel(ButtonActionListener buttonActionListener) {
31          Dimension d = new Dimension(40, 10);
32          
33          toClearLabel = new JLabel("-1", JLabel.CENTER);
34          toClearLabel.setPreferredSize(d);
35          toClearLabel.setName("toClearLabel");
36          add(toClearLabel, BorderLayout.WEST);
37          
38          JButton reStartButton = new JButton("New game");
39          reStartButton.setName("NewButton");
40          reStartButton.addActionListener(buttonActionListener);
41          add(reStartButton, BorderLayout.CENTER);
42          
43          clickCountLabel = new JLabel("0", JLabel.CENTER);
44          clickCountLabel.setPreferredSize(d);
45          add(clickCountLabel, BorderLayout.EAST);
46      }
47  
48      /**
49       * Update the action label with the given number.
50       * 
51       * @param actionCount  the action count to display
52       */
53      public void setActions(int actionCount) {
54          clickCountLabel.setText(actionCount + "");
55      }
56  
57      /**
58       * Update the remaining tiles count with the given number.
59       * 
60       * @param remainingTiles  the tile coint to display
61       */
62      public void setRemainingCount(int remainingTiles) {
63          toClearLabel.setText(remainingTiles + "");
64      }
65  }
